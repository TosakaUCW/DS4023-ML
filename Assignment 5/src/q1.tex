\section*{Question 1}

\subsection*{Data}
\[
    X=\{(0,2),(0,0),(1,0),(5,0),(5,2)\}.
\]

\subsection*{Objective}
Let $c_i\in\{1,2\}$ be the cluster assignment of $x_i$ and $\mu_k$ be the centroid of cluster $k$.
K-means minimizes the within-cluster sum of squares (WCSS):
\[
    J(\{\mu_k\},\{c_i\})=\sum_{i=1}^{N}\|x_i-\mu_{c_i}\|^2.
\]

\subsection*{Algorithm (Derivations)}
\paragraph{Assignment step.}
For fixed centroids $\{\mu_k\}$, each point is assigned to the nearest centroid:
\[
    c_i=\arg\min_{k\in\{1,2\}}\|x_i-\mu_k\|^2.
\]

\paragraph{Update step.}
For fixed assignments, minimize
\[
    J_k(\mu_k)=\sum_{i:c_i=k}\|x_i-\mu_k\|^2.
\]
Differentiate and set to zero:
\[
    \frac{\partial J_k}{\partial \mu_k}=2\sum_{i:c_i=k}(\mu_k-x_i)=0
    \quad\Rightarrow\quad
    \mu_k=\frac{1}{|S_k|}\sum_{i\in S_k}x_i,
\]
where $S_k=\{i:c_i=k\}$.

\subsection*{Results}
\[
    S_1=\{(0,2),(0,0),(1,0)\},\qquad
    S_2=\{(5,0),(5,2)\}.
\]
Centroids:
\[
    \mu_1=\frac{(0,2)+(0,0)+(1,0)}{3}=\left(\frac{1}{3},\frac{2}{3}\right),\qquad
    \mu_2=\frac{(5,0)+(5,2)}{2}=(5,1).
\]
WCSS:
\[
    J=\sum_{x\in S_1}\|x-\mu_1\|^2+\sum_{x\in S_2}\|x-\mu_2\|^2
    =\frac{40}{3}\approx 13.3333.
\]
