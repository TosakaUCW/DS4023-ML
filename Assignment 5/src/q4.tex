\section*{Question 4}

\subsection*{Data matrix}
\[
    X=
    \begin{pmatrix}
        1 & 2 & 3 & 4 \\
        5 & 5 & 6 & 7 \\
        1 & 4 & 2 & 3 \\
        5 & 3 & 2 & 1 \\
        8 & 1 & 2 & 2
    \end{pmatrix}\in\mathbb{R}^{5\times 4}.
\]

\subsection*{Derivation and procedure}
\paragraph{(1) Centering.}
Let column mean be $\bar{\bm{x}}=\frac{1}{n}\sum_{i=1}^n \bm{x}_i$.
Centered data:
\[
    X_c = X - \bm{1}\bar{\bm{x}}^{\top}.
\]

\paragraph{(2) Covariance matrix.}
\[
    S=\frac{1}{n-1}X_c^\top X_c \in \mathbb{R}^{4\times 4}.
\]

\paragraph{(3) Eigen-decomposition.}
Solve
\[
    S\bm{v}_j=\lambda_j \bm{v}_j,\qquad \lambda_1\ge \lambda_2\ge \lambda_3\ge \lambda_4.
\]
Take the top two eigenvectors:
\[
    W = [\bm{v}_1\ \bm{v}_2]\in\mathbb{R}^{4\times 2}.
\]

\paragraph{(4) Projection to 2D.}
\[
    Z = X_c W \in\mathbb{R}^{5\times 2}.
\]
